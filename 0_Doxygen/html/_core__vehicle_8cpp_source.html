247void ACore_vehicle::Check_for_door_pos()
248{
249 e_seat_type seat_type = e_seat_type::NONE;
250 float arr_vec_distance[4]{ 0.f };
251 bool is_not_near_door = false;
252
253 // 벡터 확인
254 for (int i = 0; i < m_vehicle_data.max_seater; i++)
255 {
256 FVector player_pos = m_player->GetActorLocation();
257 FVector car_door_pos = m_map_door_pos[(e_seat_type)i];
258 arr_vec_distance[i] = FVector::Distance(player_pos.GetAbs(), car_door_pos.GetAbs());
259 }
260 // 문짝이 2개일 시
261 if (m_vehicle_data.max_seater == 2)
262 {
263 // 왼쪽 첫번째 좌석
264 if (arr_vec_distance[0] < arr_vec_distance[1])
265 seat_type = e_seat_type::FIRST;
266
267 // 왼쪽 두번째 좌석
268 else if (arr_vec_distance[1] < arr_vec_distance[0])
269 seat_type = e_seat_type::SECOND;
270 }
271 // 문짝이 4개일 시
272 else
273 {
274 // 왼쪽 첫번째 좌석
275 if (arr_vec_distance[0] < arr_vec_distance[1] &&
276 arr_vec_distance[0] < arr_vec_distance[2] &&
277 arr_vec_distance[0] < arr_vec_distance[3])
278 seat_type = e_seat_type::FIRST;
279
280 // 왼쪽 두번째 좌석
281 else if (arr_vec_distance[1] < arr_vec_distance[0] &&
282 arr_vec_distance[1] < arr_vec_distance[2] &&
283 arr_vec_distance[1] < arr_vec_distance[3])
284 seat_type = e_seat_type::SECOND;
285
286 // 오른쪽 첫번째 좌석
287 else if (arr_vec_distance[2] < arr_vec_distance[3] &&
288 arr_vec_distance[2] < arr_vec_distance[0] &&
289 arr_vec_distance[2] < arr_vec_distance[1])
290 seat_type = e_seat_type::THIRD;
291
292 // 오른쪽 첫번째 좌석
293 else if (arr_vec_distance[3] < arr_vec_distance[0] &&
294 arr_vec_distance[3] < arr_vec_distance[1] &&
295 arr_vec_distance[3] < arr_vec_distance[2])
296 seat_type = e_seat_type::FOURTH;
297 }
298 if (seat_type != e_seat_type::NONE)
299 {
300 m_map_empty_seat[seat_type] = true;
301 m_player->current_seat_type = seat_type;
302 }
303}
