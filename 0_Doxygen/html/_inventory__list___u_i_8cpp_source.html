146FVector2D UInventory_list_UI::Get_distance_between_slot_cursor()
147{
148 // 마우스 위치를 구함
149 FVector2D mouse_pos = UWidgetLayoutLibrary::GetMousePositionOnViewport(GetWorld());
150
151 if (p_slot_obj)
152 {
153 // 현재 이미지 위치를 구함
154 FVector2D img_pos = FVector2D::ZeroVector, dummy_vec;
155 auto cached_geometry = p_slot_obj->GetCachedGeometry();
156 USlateBlueprintLibrary::LocalToViewport(GetWorld(), cached_geometry, FVector2D::ZeroVector, dummy_vec, img_pos);
157
158 // 마우스 좌표 - 이미지 좌표 간 거리
159 float posX = 0.f, posY = 0.f;
160
161 if (img_pos.X > mouse_pos.X)
162 posX = img_pos.X - mouse_pos.X;
163
164 else
165 posX = mouse_pos.X - img_pos.X;
166
167 if (img_pos.Y > mouse_pos.Y)
168 posY = img_pos.Y - mouse_pos.Y;
169
170 else
171 posY = mouse_pos.Y - img_pos.Y;
172
173 return FVector2D(posX, posY);
174 }
175 return FVector2D::ZeroVector;
176}
