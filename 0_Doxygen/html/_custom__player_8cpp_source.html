149void ACustom_player::Check_if_moving()
150{
151 // 움직이고 있지 않음
152 if (GetVelocity().IsZero())
153 {
154 if (m_is_moving)
155 {
156 if (current_state == e_player_state::CROUCH_WALK)
157 current_state = e_player_state::CROUCH;
158
159 else if (current_state == e_player_state::PRONING_WALK)
160 current_state = e_player_state::PRONING;
161
162 else if (current_state == e_player_state::AIM_WALK)
163 current_state = e_player_state::AIM;
164
165 else
166 current_state = e_player_state::IDLE;
167 }
168 m_is_moving = false;
169 is_sprinting = false;
170 }
171 else
172 {
173 // 숙이고 있음
174 if (current_state == e_player_state::CROUCH ||
175 current_state == e_player_state::CROUCH_AIM)
176 current_state = e_player_state::CROUCH_WALK;
177
178 // 엎드리고 있음
179 else if (current_state == e_player_state::PRONING)
180 current_state = e_player_state::PRONING_WALK;
181
182 else if (current_state == e_player_state::AIM)
183 current_state = e_player_state::AIM_WALK;
184
185 else
186 {
187 // 떨어지고 있음
188 if (GetCharacterMovement()->IsFalling())
189 {
190 // 뛰면서 점프후 착지
191 if (m_sprint_multiplier > 1.f)
192 {
193 is_sprinting = true;
194 current_state = e_player_state::SPRINT_JUMP;
195 }
196 else // 점프함
197 current_state = e_player_state::JUMP;
198 }
199 else // 지면에 닿고있음
200 {
201 if (current_state == e_player_state::SPRINT_JUMP)
202 current_state = e_player_state::SPRINT;
203
204 // 뛰고있음
205 else if (current_state == e_player_state::SPRINT)
206 {
207 if (current_oxygen < 0)
208 {
209 current_state = e_player_state::IDLE;
210 m_sprint_multiplier = 1;
211 GetCharacterMovement()->MaxWalkSpeed = 350.f;
212 is_sprinting = false;
213 return;
214 }
215 is_sprinting = true;
216
217 if (m_sprint_multiplier < 1.75f)
218 {
219 m_sprint_multiplier += 0.25f;
220 GetCharacterMovement()->MaxWalkSpeed *= m_sprint_multiplier;
221 }
222 }
223 //
224 else if (current_state == e_player_state::IDLE ||
225 current_state == e_player_state::JUMP)
226 current_state = e_player_state::WALK;
227 }
228 }
229 m_is_moving = true;
230 }
231}

233void ACustom_player::Check_for_object()
234{
235 FVector direction = GetActorForwardVector() * 50;
236 FVector begin_pos = GetMesh()->GetSocketLocation("detect_object_ray_sock");
237 FVector end_pos = begin_pos + direction;
238 FHitResult hit_result;
239 bool is_collided = false;
240
241 GetWorld()->LineTraceSingleByProfile(hit_result, begin_pos, end_pos, "Object");
242
243 AActor* hitted_actor = hit_result.GetActor();
244
245 // 충돌한 오브젝트가 있을 시
246 if (hitted_actor)
247 {
248 // 충돌한 오브젝트가 무기일 시
249 if (hitted_actor->IsA<ACore_weapon>() ||
250 hitted_actor->IsA<ACore_throwable_weapon>() ||
251 hitted_actor->IsA<ACore_melee_weapon>())
252 {
253 is_collided = true;
254 mp_collided_weapon = hitted_actor;
255 }
256 if (auto p_obj = Cast<ABase_interaction>(hitted_actor))
257 p_obj->is_player_near = true;
258 }
259 else
260 {
261 // 무기 접근 후 벗어남
262 if (mp_collided_weapon)
263 {
264 Cast<ABase_interaction>(mp_collided_weapon)->is_player_near = false;
265 mp_collided_weapon = nullptr;
266 }
267 }
268}

270void ACustom_player::Check_if_is_vehicle_near()
271{
272 if (is_in_vehicle)
273 return;
274
275 APlayerCameraManager* camera_manager = UGameplayStatics::GetPlayerCameraManager(this, 0);
276 FVector begin_pos = GetActorLocation();
277 FVector forward_vec = GetActorForwardVector() * 50;
278 FHitResult hit_result;
279 FVector end_pos = begin_pos + forward_vec;
280 auto p_world = GetWorld();
281 p_world->LineTraceSingleByObjectType(hit_result, begin_pos, end_pos, FCollisionObjectQueryParams(ECC_Vehicle));
282 //DrawDebugLine(p_world, begin_pos, end_pos, FColor::Red, true, 5.f, (uint8)0U, 1.f);
283
284 // 차량을 감지
285 AActor* hitted_actor = hit_result.GetActor();
286
287 if (!hitted_actor)
288 {
289 if (mp_collided_vehicle)
290 {
291 mp_collided_vehicle->is_collided = false;
292 mp_collided_vehicle = nullptr;
293 }
294 return;
295 }
296 if (hitted_actor->IsA(ACore_vehicle::StaticClass()))
297 {
298 mp_collided_vehicle = Cast<ACore_vehicle>(hitted_actor);
299
300 if (mp_collided_vehicle)
301 mp_collided_vehicle->is_collided = true;
302 }

365void ACustom_player::Custom_crouch()
366{
367 if (m_is_moving ||
368 GetCharacterMovement()->IsFalling())
369 return;
370
371 p_spring_arm_comp->SetRelativeLocation(FVector(0.f, 0.f, 80.f));
372
373 // 숙이고 있음
374 if (GetCharacterMovement()->IsCrouching())
375 {
376 if (current_state == e_player_state::CROUCH_AIM)
377 current_state = e_player_state::AIM;
378
379 else
380 current_state = e_player_state::IDLE;
381
382 UnCrouch();
383 }
384 else
385 {
386 if (current_state == e_player_state::AIM)
387 current_state = e_player_state::CROUCH_AIM;
388
389 else
390 current_state = e_player_state::CROUCH;
391
392 Crouch();
393 }

396void ACustom_player::Proning()
397{

400 case e_player_state::IDLE:
401 current_state = e_player_state::PRONING;
402 p_spring_arm_comp->SetRelativeLocation(FVector(0.f, 0.f, -50.f));

404
405 case e_player_state::CROUCH:
406 case e_player_state::CROUCH_WALK:
407 current_state = e_player_state::PRONING;
408 p_spring_arm_comp->SetRelativeLocation(FVector(0.f, 0.f, -50.f));
409 UnCrouch();
410 break;
411
412 case e_player_state::AIM:
413 current_state = e_player_state::PRONING_AIM;
414 break;
415
416 case e_player_state::PRONING:
417 case e_player_state::PRONING_WALK:
418 current_state = e_player_state::IDLE;
419 p_spring_arm_comp->SetRelativeLocation(FVector(0.f, 0.f, 80.f));

466void ACustom_player::Aim()
467{
468 bool is_first_equipped = false;
469 bool is_second_equipped = false;
470 int code = 0;
471
472 // 무기 장착하고 있지 않은 상태
473 if (!is_first_equipped &&
474 !is_second_equipped)
475 return;
476
477 switch (current_state)
478 {
479 // 기본자세 > 에임
480 case e_player_state::IDLE: current_state = e_player_state::AIM; code = 1; break;
481 case e_player_state::CROUCH: current_state = e_player_state::CROUCH_AIM; code = 1; break;
482 case e_player_state::PRONING: current_state = e_player_state::PRONING_AIM; code = 1; break;
483
484 // 에임 > 기본 자세
485 case e_player_state::AIM: current_state = e_player_state::IDLE; code = 0; break;
486 case e_player_state::CROUCH_AIM: current_state = e_player_state::CROUCH; code = 0; break;
487 case e_player_state::PRONING_AIM: current_state = e_player_state::PRONING; code = 0; break;
488 }
489 is_aiming = (bool)code;
490 //Changing_aim_pose(code);
491}
