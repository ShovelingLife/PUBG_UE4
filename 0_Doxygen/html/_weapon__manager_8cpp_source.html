119void AWeapon_manager::Attach_weapon(ABase_interaction* _p_tmp_weapon, FString _socket_name)
120{
121 auto current_player_mesh = UGameplayStatics::GetPlayerCharacter(GetWorld(), 0)->GetMesh();
122
123 // 소켓 기반 무기 종류 판별 후 다운캐스팅

125 {

130 {

140 current_weapon_type = e_current_weapon_type::SECOND;
141 }
142 }
143 // 투척류/근접무기
144 else
145 {
146 if (_p_tmp_weapon->IsA<ACore_throwable_weapon>())
147 {
148 p_throwable = Cast<ACore_throwable_weapon>(_p_tmp_weapon);
149 current_weapon_type = e_current_weapon_type::THROWABLE;
150 }
151 else if (_p_tmp_weapon->IsA<ACore_melee_weapon>())

163
164 else if (p_static_mesh_comp)
165 p_static_mesh_comp->AttachToComponent(current_player_mesh, FAttachmentTransformRules::SnapToTargetNotIncludingScale, *_socket_name);
166
167 // 무기 오브젝트를 인벤토리 매니저에 부착

171void AWeapon_manager::Reset_weapon_after_detaching(ABase_interaction* _weapon, FTransform _new_pos)
172{
173 USkeletalMeshComponent* p_skeletal_mesh_comp = _weapon->p_skeletal_mesh_comp;
174 UStaticMeshComponent* p_static_mesh_comp = _weapon->p_static_mesh_comp;
175
176 // 컴포넌트를 탈착 > 현재 루트 컴포넌트에 부착 > 트랜스폼 초기화
177 if (p_skeletal_mesh_comp)
178 {
179 p_skeletal_mesh_comp->DetachFromComponent(FDetachmentTransformRules::KeepRelativeTransform);
180 p_skeletal_mesh_comp->AttachToComponent(_weapon->GetRootComponent(), FAttachmentTransformRules::KeepRelativeTransform);
181 p_skeletal_mesh_comp->ResetRelativeTransform();
182 }
183 else if (p_static_mesh_comp)
184 {
185 p_static_mesh_comp->DetachFromComponent(FDetachmentTransformRules::KeepRelativeTransform);
186 p_static_mesh_comp->AttachToComponent(_weapon->GetRootComponent(), FAttachmentTransformRules::KeepRelativeTransform);
187 p_static_mesh_comp->ResetRelativeTransform();
188 }
189 // 현재 무기를 탈착 후 월드에 소환
190 _weapon->DetachFromActor(FDetachmentTransformRules::KeepWorldTransform);
191 _weapon->SetActorTransform(_new_pos);

308void AWeapon_manager::Reload()
309{
310 ABase_interaction* tmp_weapon = Get_weapon(current_weapon_type);
311
312
313 if (auto p_gun = Cast<ACore_weapon>(tmp_weapon))
314 {
315 int result = 0;
316 m_is_reloading = true;
317 auto weapon_data = p_gun->weapon_data;
318
319 // 허용 총알 개수가 똑같을 시
320 if (weapon_data.current_bullet_count == weapon_data.max_bullet_count)
321 return;
322
323 // 중간에 장전
324 else if (weapon_data.current_bullet_count > 0)
325 result = weapon_data.max_bullet_count - weapon_data.current_bullet_count;
326
327 // 전체 총알 소모 후 장전
328 else
329 result = p_gun->weapon_data.max_bullet_count;
330
331 weapon_data.current_max_bullet_count -= result;
332 weapon_data.current_bullet_count += result;
333 Play_sound(e_weapon_sound_type::RELOAD);
334 }

337bool AWeapon_manager::Scroll_select(int _pos)
338{
339 // 현재 무기 인덱스 갖고와서 선택
340 e_current_weapon_type final_index = e_current_weapon_type::NONE;
341 int current_index = (int)current_weapon_type;
342 int total_weapon = -1;
343
344 for (int i = 0; i < 5; i++)
345 {
346 // 현재 장착 중인 무기
347 if (arr_is_weapon_equipped[i])
348 total_weapon++;
349 }
350 // 무기가 1개일 시 옮길 필요가 없음
351 if (total_weapon == 0)
352 return false;
353
354 // 아래로 스크롤
355 if (_pos == -1)
356 {
357 // 마지막 원소 0 도달 시
358 if (current_index == (int)e_current_weapon_type::FIRST)
359 current_weapon_type = Find_weapon_index("down", (int)e_current_weapon_type::THROWABLE);
360
361 else
362 {
363 // 현재 위치에서 탐색
364 final_index = Find_weapon_index("down", current_index - 1);
365
366 // 발견하지 못함
367 if (final_index == e_current_weapon_type::NONE)
368 current_weapon_type = Find_weapon_index("down", (int)e_current_weapon_type::THROWABLE);
369
370 else
371 current_weapon_type = final_index;
372 }
373 }
374 // 위로 스크롤
375 else
376 {
377 // 현재 마지막 원소에 접근 할 시
378 if (current_index == (int)e_current_weapon_type::THROWABLE)
379 {
380 if (p_first_gun)
381 current_weapon_type = e_current_weapon_type::FIRST;
382
383 else
384 current_weapon_type = Find_weapon_index("up", (int)e_current_weapon_type::FIRST);
385 }
386 else
387 {
388 // 현재 위치에서 탐색
389 final_index = Find_weapon_index("up", current_index + 1);
390
391 // 발견하지 못함
392 if (final_index == e_current_weapon_type::NONE)
393 current_weapon_type = Find_weapon_index("up", (int)e_current_weapon_type::FIRST);
394
395 else
396 current_weapon_type = final_index;
397 }
398 }
399 return true;
400}

408void AWeapon_manager::Change_shoot_mode()
409{
410 auto p_weapon = Get_weapon(current_weapon_type);
411 int max_gun_shoot_type = 0;
412
413 if (!p_weapon)
414 return;
415
416 // 총기일 시
417 if (p_weapon->IsA<ACore_weapon>())
418 {
419 auto p_gun = Cast<ACore_weapon>(p_weapon);
420 FString weapon_group = p_gun->weapon_data.weapon_group_type;
421
422 // 단발/점사/연사 가능
423 if (weapon_group == "Assault" ||
424 weapon_group == "SMG")
425 max_gun_shoot_type = (int)e_gun_shoot_type::CONSECUTIVE;
426
427 // 단발/점사 가능
428 else if (weapon_group == "Shotgun")
429 max_gun_shoot_type = (int)e_gun_shoot_type::BURST;
430
431 // 오직 단발만 가능
432 else if (weapon_group == "Handgun" ||
433 weapon_group == "Sniper" ||
434 weapon_group == "Special")
435 max_gun_shoot_type = (int)e_gun_shoot_type::SINGLE;
436
437 // 격발 방식 변경
438 int current_gun_shoot_type = (int)p_gun->gun_shoot_type;
439
440 if ((int)p_gun->gun_shoot_type < max_gun_shoot_type)
441 p_gun->gun_shoot_type = (e_gun_shoot_type)++current_gun_shoot_type;
442
443 // 초기화
444 else if ((int)p_gun->gun_shoot_type == max_gun_shoot_type)
445 p_gun->gun_shoot_type = e_gun_shoot_type::SINGLE;
446 }
447}
